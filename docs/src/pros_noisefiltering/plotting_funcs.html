 20def plot_comparative_response(wt_obj,   # cutoff frequency
 21                              filter_func, response_offset=2e-4,
 22                              Kolmogorov_offset=1,
 23                              figsize=(16, 9),
 24                              nperseg=1024,
 25                              xlim=[1e1, 1e5],
 26                              ylim=[1e-8, 1e-1],
 27                              plot_th=False):
 28    """#TOD make this part of WT_NoiProc.
 29
 30    Args:
 31    - wt_obj (Graph_data_container): List of graph containers
 32    - response_offset (float, optional): Setting a response offset.
 33        - Defaults to 2e-4.
 34    - figsize (tuple, optional): Determining the size of the produced figure.
 35        - Defaults to (16,9).
 36    """
 37    sig = wt_obj.data
 38    fs_Hz = wt_obj.fs_Hz
 39
 40    filt_p = filter_func.params
 41    sos = signal.butter(N=filt_p['filter order'], Wn=filt_p['fc_Hz'],
 42                        btype='lp', fs=fs_Hz, output='sos')
 43
 44    filtered = filter_func(sig, fs_Hz)
 45
 46    # calculate spectrum
 47    f, Pxx_spec = signal.welch(sig, fs_Hz,
 48                               window='flattop', nperseg=nperseg,
 49                               scaling='density')
 50    f, Pxx_spec_filt = signal.welch(filtered, fs_Hz,
 51                                    window='flattop', nperseg=nperseg,
 52                                    scaling='density')
 53
 54    # #TODO: this is for testing
 55    # Pxx_spec= Pxx_spec**2
 56    # Pxx_spec_filt = Pxx_spec_filt**2
 57
 58    if plot_th:
 59        t = np.arange(0, len(sig), 1,
 60                      dtype='int') / fs_Hz
 61        # plot time domain
 62        fig1, (ax1, ax2) = plt.subplots(2, 1,
 63                                        sharex=True, sharey=True,
 64                                        figsize=figsize)
 65        fig1.suptitle(
 66            'Time Domain Filtering of signal with f1=10[Hz], f2=20[Hz] and noise')
 67        ax1.plot(t, sig)
 68        ax1.set_title('raw signal')
 69        # ax1.axis([0, 1, -2, 2])
 70        ax2.plot(t, filtered)
 71        ax2.set_title('After filter')
 72        # ax2.axis([0, 1, -2, 2])
 73        ax2.set_xlabel('Time [seconds]')
 74        plt.tight_layout()
 75
 76    wb, hb = signal.sosfreqz(sos, worN=np.logspace(start=1, stop=5),
 77                             whole=True, fs=fs_Hz)
 78    fb = wb   # /(2*np.pi) # convert rad/s to Hz
 79
 80    fig2, ax2 = plt.subplots(1, 1, sharex=True,
 81                             figsize=figsize)
 82    ax2.plot(fb,
 83             response_offset*abs(np.array(hb)), '--', lw=3,
 84             label='filter response')
 85    ax2.semilogy(f, np.sqrt(Pxx_spec), '.', label='raw')
 86    ax2.semilogy(f, np.sqrt(Pxx_spec_filt), '.', label='filtered')
 87    if Kolmogorov_offset is not None:
 88        KOLMOGORV_CONSTANT = - 5.0/3
 89        xs = f[1:]
 90        ys = xs**(KOLMOGORV_CONSTANT)*Kolmogorov_offset
 91        ax2.plot(xs, ys, 'r--', label='Kolmogorov -5/3')
 92
 93    ax2.set_title('Filter frequency response (cutoff: {}Hz) -  {}'.format(
 94        filter_func.params.get('fc_Hz', None),
 95        wt_obj.description))
 96    ax2.set_xlabel('Frequency [Hz]')
 97    ax2.set_ylabel('Power Spectrum density [V**2/Hz]')
 98    ax2.set_xscale('log')
 99    ax2.set_yscale('log')
100    ax2.margins(0, 0.1)
101    ax2.grid(which='both', axis='both')
102    ax2.set_xlim(xlim)
103    ax2.set_ylim(ylim)
104    ax2.legend()

29
30        Args:
31        - ds (np.ndarray): data signal object
32        - fs_Hz (float): sampling frequency
33        - fc_Hz (int, optional): cutoff frequency of filter(low pass def:100)

36

45        return filtered
